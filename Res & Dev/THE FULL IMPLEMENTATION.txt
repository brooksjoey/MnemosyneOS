```bash
confirm() {
    if [ "$ASSUME_YES" = true ]; then
        return 0
    fi
    
    local prompt="$1"
    local response
    
    read -rp "$prompt" response
    case "$response" in
        [yY][eE][sS]|[yY]) 
            return 0
            ;;
        *)
            return 1
            ;;
    esac
}
```

**src/mnemosyneos/main.py**
```python
#!/usr/bin/env python3
"""
MnemosyneOS Main Application Entry Point
A reflective, self-healing persistent memory service
"""

import asyncio
import logging
from contextlib import asynccontextmanager
from typing import AsyncGenerator

from fastapi import FastAPI, Request, status
from fastapi.exceptions import RequestValidationError
from fastapi.responses import JSONResponse
from prometheus_client import make_asgi_app

from mnemosyneos.core.config import settings
from mnemosyneos.core.logging import configure_logging
from mnemosyneos.core.security import setup_security_middleware
from mnemosyneos.db.session import init_db, close_db, verify_integrity
from mnemosyneos.api.v1.router import api_router
from mnemosyneos.tasks.celery_app import celery_app, init_celery

# Configure logging
configure_logging()

logger = logging.getLogger(__name__)

@asynccontextmanager
async def lifespan(app: FastAPI) -> AsyncGenerator[None, None]:
    """Application lifespan management"""
    # Startup
    logger.info("Starting MnemosyneOS service")
    
    try:
        # Initialize database
        await init_db()
        
        # Verify data integrity
        await verify_integrity()
        
        # Initialize Celery
        await init_celery()
        
        logger.info("MnemosyneOS startup completed successfully")
        yield
        
    except Exception as e:
        logger.critical(f"Failed to start MnemosyneOS: {e}")
        raise
    
    finally:
        # Shutdown
        logger.info("Shutting down MnemosyneOS service")
        await close_db()

# Create FastAPI application
app = FastAPI(
    title="MnemosyneOS",
    description="Reflective, self-healing persistent memory service",
    version="0.1.0",
    lifespan=lifespan,
    docs_url="/docs" if settings.debug else None,
    redoc_url="/redoc" if settings.debug else None,
)

# Add security middleware
setup_security_middleware(app)

# Add Prometheus metrics
metrics_app = make_asgi_app()
app.mount("/metrics", metrics_app)

# Include API routers
app.include_router(api_router, prefix="/v1")

# Health check endpoints
@app.get("/healthz")
async def health_check():
    """Kubernetes health check endpoint"""
    return {"status": "healthy"}

@app.get("/readyz")
async def readiness_check():
    """Kubernetes readiness endpoint"""
    # Add database connection check
    return {"status": "ready"}

@app.get("/version")
async def version_info():
    """Return service version information"""
    return {
        "name": "MnemosyneOS",
        "version": "0.1.0",
        "description": "Reflective, self-healing persistent memory service"
    }

# Exception handlers
@app.exception_handler(RequestValidationError)
async def validation_exception_handler(request: Request, exc: RequestValidationError):
    """Handle validation errors"""
    return JSONResponse(
        status_code=status.HTTP_422_UNPROCESSABLE_ENTITY,
        content={"detail": exc.errors(), "body": exc.body},
    )

@app.exception_handler(Exception)
async def general_exception_handler(request: Request, exc: Exception):
    """Handle general exceptions"""
    logger.error(f"Unhandled exception: {exc}", exc_info=True)
    return JSONResponse(
        status_code=status.HTTP_500_INTERNAL_SERVER_ERROR,
        content={"detail": "Internal server error"},
    )

if __name__ == "__main__":
    import uvicorn
    uvicorn.run(
        app,
        host=settings.api_host,
        port=settings.api_port,
        log_config=None,  # Use our structured logging
    )
```

```python
from fastapi import APIRouter, Depends, HTTPException, status, Query, BackgroundTasks
from fastapi.security import HTTPBearer, HTTPAuthorizationCredentials

from mnemosyneos.core.security import verify_api_key
from mnemosyneos.core.config import settings
from mnemosyneos.db.session import get_db_session
from mnemosyneos.memory.manager import MemoryManager
from mnemosyneos.memory.graph import BeliefGraph
from mnemosyneos.tasks.celery_app import ingest_memory_task, reflect_task, compress_task
from mnemosyneos.utils.backup import run_backup, run_restore

logger = logging.getLogger(__name__)
router = APIRouter()
security = HTTPBearer()

# Dependency to get memory manager
async def get_memory_manager():
    """Get memory manager instance"""
    async with get_db_session() as session:
        manager = MemoryManager(session)
        yield manager

@router.post("/remember", status_code=status.HTTP_202_ACCEPTED)
async def remember_memory(
    memory_data: dict,
    background_tasks: BackgroundTasks,
    credentials: HTTPAuthorizationCredentials = Depends(security),
    manager: MemoryManager = Depends(get_memory_manager)
):
    """Ingest a new memory"""
    verify_api_key(credentials.credentials)
    
    # Validate memory data
    if not memory_data.get("content"):
        raise HTTPException(
            status_code=status.HTTP_400_BAD_REQUEST,
            detail="Memory content is required"
        )
    
    # Queue for async processing
    background_tasks.add_task(
        ingest_memory_task.delay,
        memory_data,
        credentials.credentials
    )
    
    return {"status": "accepted", "message": "Memory queued for processing"}

@router.get("/recall")
async def recall_memory(
    query: str,
    k: int = Query(5, ge=1, le=100),
    hybrid: bool = Query(True),
    credentials: HTTPAuthorizationCredentials = Depends(security),
    manager: MemoryManager = Depends(get_memory_manager)
):
    """Recall memories based on query"""
    verify_api_key(credentials.credentials)
    
    try:
        results = await manager.recall(
            query=query,
            limit=k,
            hybrid_search=hybrid
        )
        return {
            "query": query,
            "results": results,
            "count": len(results)
        }
    except Exception as e:
        logger.error(f"Recall failed: {e}")
        raise HTTPException(
            status_code=status.HTTP_500_INTERNAL_SERVER_ERROR,
            detail="Failed to recall memories"
        )

@router.get("/provenance/{memory_id}")
```
